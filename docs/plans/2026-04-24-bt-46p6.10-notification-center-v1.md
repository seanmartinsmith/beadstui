## Design (locked — do not re-litigate)

- `!` opens modal on **alerts tab** (today's behavior; no regression)
- `1` opens modal on **notifications tab**
- If modal is already open, either key switches to that tab (no close + reopen)
- Pressing the same key that opened the current tab closes the modal
- `tab` cycles tabs when modal is open
- Per-tab cursor + filter state preserved across flips
- v1 notifications = `events.RingBuffer.Snapshot()` reversed to newest-first, dismissed hidden
- Modal header shows both counts: active tab bracketed, inactive dimmed

// visibleNotifications returns ring-buffer events filtered by dismissed state,
// newest-first. v1 hides dismissed; bt-46p6.13 may expose them via filter.
func (m Model) visibleNotifications() []events.Event {
    snap := m.events.Snapshot()
    out := make([]events.Event, 0, len(snap))
    // Snapshot is oldest-first; reverse to newest-first.
    for i := len(snap) - 1; i >= 0; i-- {
        if snap[i].Dismissed {
            continue
        }
        out = append(out, snap[i])
    }
    return out
}
```

// handleNotificationsKey routes keypresses when the shared modal is on the
// notifications tab (bt-46p6.10). Mirrors the alerts-tab handler shape but
// uses ring-buffer Dismiss() instead of the dismissedAlerts map.
func (m Model) handleNotificationsKey(msg tea.KeyPressMsg) (Model, tea.Cmd) {
    activeNotifs := m.visibleNotifications()
    switch msg.String() {
    case "j", "down":
        if m.notificationsCursor < len(activeNotifs)-1 {
            m.notificationsCursor++
        }
        return m, nil
    case "k", "up":
        if m.notificationsCursor > 0 {
            m.notificationsCursor--
        }
        return m, nil
    case "enter":
        if m.notificationsCursor < len(activeNotifs) {
            beadID := activeNotifs[m.notificationsCursor].BeadID
            if beadID != "" {
                if m.workspaceMode && m.activeRepos != nil {
                    if issue, ok := m.data.issueMap[beadID]; ok {
                        repoKey := IssueRepoKey(*issue)
                        if repoKey != "" && !m.activeRepos[repoKey] {
                            m.activeRepos[repoKey] = true
                            m.applyFilter()
                        }
                    }
                }
                for i, item := range m.list.Items() {
                    if it, ok := item.(IssueItem); ok && it.Issue.ID == beadID {
                        m.list.Select(i)
                        break
                    }
                }
            }
        }
        m.closeModal()
        return m, nil
    case "c":
        if m.notificationsCursor < len(activeNotifs) {
            m.events.Dismiss(activeNotifs[m.notificationsCursor].ID)
            remaining := len(m.visibleNotifications())
            if m.notificationsCursor >= remaining {
                m.notificationsCursor = remaining - 1
            }
            if m.notificationsCursor < 0 {
                m.notificationsCursor = 0
            }
            if remaining == 0 {
                m.closeModal()
            }
        }
        return m, nil
    case "C":
        m.events.DismissAll()
        m.notificationsCursor = 0
        m.closeModal()
        return m, nil
    case "esc", "q", "!", "1":
        m.closeModal()
        return m, nil
    }
    return m, nil
}
```

// formatNotificationRow renders a single ring-buffer event as a one-line
// notification. Format: "15:04  closed     bt-46p6.1   Fix: modal expands…"
// Widths keep the columns aligned; title is truncated at runtime width.
func formatNotificationRow(e events.Event, width int) string {
    timeStr := e.At.Format("15:04")
    kindStr := padRight(e.Kind.String(), 9)   // "created", "edited", "closed", "commented", "bulk"
    idStr := padRight(e.BeadID, 12)
    // 2 spaces after time, 1 space after kind/id, 2-space gap before title:
    //   "hh:mm" + "  " + kind(9) + " " + id(12) + "  " + title
    consumed := len(timeStr) + 2 + 9 + 1 + 12 + 2
    titleWidth := width - consumed
    if titleWidth < 10 {
        titleWidth = 10
    }
    return timeStr + "  " + kindStr + " " + idStr + "  " + truncate(e.Title, titleWidth)
}
```

// renderNotificationsTab builds the notifications tab body. Mirrors the
// outer shape of renderAlertsTab (summary line, scrollable page of rows,
// pagination hints) but reads from events.RingBuffer and has no filter UI
// in v1. Dismissed events are hidden (see visibleNotifications).
func (m Model) renderNotificationsTab() string {
    t := m.theme
    panelWidth := min(80, m.width-4)
    innerWidth := panelWidth - 4

    active := m.visibleNotifications()

    var sb strings.Builder

    if len(active) == 0 {
        sb.WriteString(lipgloss.NewStyle().Foreground(ColorSuccess).Render(" No notifications"))
        sb.WriteString("\n")
        return sb.String()
    }

    totalStyle := lipgloss.NewStyle().Foreground(t.Secondary)
    sb.WriteString(" ")
    sb.WriteString(totalStyle.Render(fmt.Sprintf("%d events", len(active))))
    sb.WriteString("\n\n")

    // Paginate using the same visible-lines helper as alerts tab.
    pageSize := m.alertsVisibleLines()
    start := (m.notificationsCursor / pageSize) * pageSize
    end := start + pageSize
    if end > len(active) {
        end = len(active)
    }

    // Above-hint
    if start > 0 {
        sb.WriteString(lipgloss.NewStyle().Foreground(t.Muted).
            Render(fmt.Sprintf(" ▴ %d more above", start)))
        sb.WriteString("\n")
    }

    cursorStyle := lipgloss.NewStyle().Foreground(t.Primary).Bold(true)
    rowStyle := lipgloss.NewStyle().Foreground(t.Base.GetForeground())

    for i := start; i < end; i++ {
        row := formatNotificationRow(active[i], innerWidth-2)
        if i == m.notificationsCursor {
            sb.WriteString(" " + cursorStyle.Render("▸ "+row))
        } else {
            sb.WriteString("   " + rowStyle.Render(row))
        }
        sb.WriteString("\n")
    }

    if end < len(active) {
        sb.WriteString(lipgloss.NewStyle().Foreground(t.Muted).
            Render(fmt.Sprintf(" ▾ %d more below", len(active)-end)))
        sb.WriteString("\n")
    }

    // Cursor indicator "N/M"
    sb.WriteString("\n")
    sb.WriteString(lipgloss.NewStyle().Foreground(t.Muted).
        Render(fmt.Sprintf("   %d/%d", m.notificationsCursor+1, len(active))))
    sb.WriteString("\n")

    // Footer hint strip (mirrors alerts footer vocabulary)
    hintStyle := lipgloss.NewStyle().Foreground(t.Muted).Italic(true)
    sb.WriteString("\n")
    sb.WriteString(hintStyle.Render("   j/k: nav  enter: open  c: dismiss  C: dismiss all  esc: close"))

    return sb.String()
}
```
